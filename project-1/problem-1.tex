\begin{minted}{matlab}
>> A = [2 1 0;1 4 3;2 4.5 6];
>> b = [2 0 -3.5]';
>> x = A\b
x =
   0.50000
   1.00000
  -1.50000
>> % Checking the answer:
>> A*x

   2.00000
   0.00000
  -3.50000
>> % Answer is b, as expected.
\end{minted}
